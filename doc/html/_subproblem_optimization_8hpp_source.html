174  std::vector<double> &x )
175 {
176  set_feasibility_thresholds ( x );

178  criticality_gradient = (*surrogate_models)[0].gradient ( x );
179  else {
180  set_zero( criticality_gradient );
181  for ( int i = 1; i < number_constraints+1; i++ )

183  (*surrogate_models)[i].gradient ( x ),
